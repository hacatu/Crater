128 inline static int cr8r_hash_ix_start(cr8r_hashtbl_t *self, const cr8r_hashtbl_ft *ft){
129  uint64_t i = self->len_a ? 64 - __builtin_clzll(self->len_a - 1) : 1;
130  uint64_t new_len = exp_primes[i];
131  void *new_table = calloc(ft->base.size, new_len);
132  if(!new_table){
133  return 0;
134  }
135  uint64_t *new_flags = calloc(sizeof(uint64_t), ((new_len - 1) >> 5) + 1);
136  if(!new_flags){
137  free(new_table);
138  return 0;
139  }
140  uint64_t new_cap = new_len*(long double)ft->load_factor;
141  //we currently have full entries and can fit new_cap before resizing. This means we must move full/(new_cap - full)
142  //entries on average per insertion. Because full == cap + 1, new_cap == new_len*load_factor, cap == len_a*load_factor,
143  //and new_len == len_a*growth_factor, full/(new_cap - full) ~~ cap/(new_cap - cap) ~~ len_a/(new_len - len_a)
144  //~~ len_a/(len_a*growth_factor - len_a) ~~ 1/(growth_factor - 1). I picked the largest prime number before each power
145  //of 2 so full/(new_cap - full) is very close to 1. It is always less than 2 so I might just use 2 instead of r
146  self->table_b = self->table_a;
147  self->table_a = new_table;
148  self->flags_b = self->flags_a;
149  self->flags_a = new_flags;
150  self->len_b = self->len_a;
151  self->len_a = new_len;
152  self->cap = new_cap;
153  self->r = ceill(self->full/((long double)new_cap - self->full));
154  self->i = 0;
155  return 1;
156 }

158 inline static void cr8r_hash_ix_move(cr8r_hashtbl_t *self, const cr8r_hashtbl_ft *ft, uint64_t n){
159  uint64_t b = self->i;
160  for(uint64_t a; b < self->len_b; ++b){
161  uint64_t f = (self->flags_b[b >> 5] >> (b&0x1F))&0x100000001ULL;
162  if(!(f&0x100000000ULL)){
163  continue;
164  }
165  void *ent = self->table_b + b*ft->base.size;
166  a = ft->hash(&ft->base,ent) % self->len_a;
167  a = cr8r_hash_find_slot_re(self, ft, a);
168  memcpy(self->table_a + a*ft->base.size, ent, ft->base.size);
169  self->flags_b[b >> 5] |= 0x1ULL << (b&0x1F);
170  self->flags_b[b >> 5] &= ~(0x100000000ULL << (b&0x1F));
171  self->flags_a[a >> 5] |= 0x100000000ULL << (a&0x1F);
172  self->flags_a[a >> 5] &= ~(0x1ULL << (a&0x1F));
173  if(!--n){
174  break;
175  }
176  }
177  self->i = b;
178  if(b == self->len_b){
179  free(self->table_b);
180  free(self->flags_b);
181  self->table_b = NULL;
182  self->flags_b = NULL;
183  self->len_b = 0;
184  self->i = 0;
185  }

248  }

251  if(!~i){
252  if(_status){
253  *_status = 0;
254  }
255  return NULL;
256  }

268  *_status = status;
269  }
